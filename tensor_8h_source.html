457 template <typename ElemType, std::size_t D, std::size_t N>
458 tensor<ElemType,D,N> Sym(const tensor<ElemType,D,N>& A)
459 {
460  tensor<ElemType,D,N> rval;
461 
462  std::array<std::size_t, N> sigma;
463  for(std::size_t i = 0; i < N; ++i)
464  {
465  //std::cout << " : " << i << std::endl;
466  sigma[i] = i;
467  }
468 
469  do {
470  for(auto curr = rval.index_begin(); curr != rval.index_end(); ++curr)
471  {
472  std::array<std::size_t, N> index = *curr;
473  for(std::size_t i = 0; i < N; ++i)
474  {
475  index[i] = (*curr)[sigma[i]];
476  }
477  rval[*curr] += A[index];
478  }

480 
481  rval *= 1.0/detail::factorial<N>::value;

485 template <std::size_t N>
486 int sgn(const std::array<std::size_t,N>& arr)
487 {
488  int rval = 1;
489  std::array<bool,N> visited;
490  visited.fill(false);
491 
492  for(int i = 1; i < N; ++i)
493  {
494  if(!visited[i])
495  {
496  std::size_t len = 0;
497  std::size_t next = i;
498  do
499  {
500  ++len;
501  visited[next] = true;
502  next = arr[next];
503  } while(!visited[next]);
504 
505  rval *= 2*(len % 2) - 1;
506  }
507  }
508  return rval;
509 }
