269 template <std::size_t level, std::size_t dimension>
270 auto getTangentH(const SurfaceMesh &mesh,
271  const tensor<double, dimension, 1> &origin,
272  SurfaceMesh::SimplexID<level> curr)
273 {
274  tensor<double, dimension, SurfaceMesh::topLevel - level> rval;
275  auto cover = mesh.get_cover(curr);
276  for (auto alpha : cover)
277  {
278  auto edge = *mesh.get_edge_up(curr, alpha);
279  const auto &v = (*mesh.get_simplex_up({alpha})).position;
280  auto next = mesh.get_simplex_up(curr, alpha);
281  rval += edge.orientation * (v-origin) * getTangentH(mesh, origin, next);
282  }
283  return rval/cover.size();
284 }

323 template <std::size_t level, std::size_t dimension>
324 auto getTangentF(const SurfaceMesh &mesh,
325  const tensor<double, dimension, 1> &origin,
326  SurfaceMesh::SimplexID<level> curr,
327  std::set<SurfaceMesh::KeyType> &cover)
328 {
329  tensor<double, dimension, SurfaceMesh::topLevel - level> rval;
330  for (auto alpha : cover)
331  {
332  auto edge = *mesh.get_edge_up(curr, alpha);
333  const auto &v = (*mesh.get_simplex_up({alpha})).position;
334  auto next = mesh.get_simplex_up(curr, alpha);
335  auto coverup = cover;
336  coverup.erase(alpha);
337  rval += edge.orientation * (v-origin) * getTangentF(mesh, origin, next, coverup);
338  }
339  return rval/cover.size();
340 }

462 template <class Inserter>
463 void selectFlipEdges(const SurfaceMesh &mesh,
464  bool preserveRidges,
465  std::function<bool(const SurfaceMesh &, const SurfaceMesh::SimplexID<2> &)> && checkFlip,
466  Inserter iter){
467  casc::NodeSet<SurfaceMesh::SimplexID<2> > ignoredEdges;
468 
469  for (auto edgeID : mesh.get_level_id<2>())
470  {
471  if ((*edgeID).selected == true){
472  if (!ignoredEdges.count(edgeID))
473  {
474  auto up = mesh.get_cover(edgeID);
475  // The mesh is not a surface mesh...
476  if (up.size() > 2)
477  {
478  // std::cerr << "This edge participates in more than 2 faces. "
479  // << "Returning..." << std::endl;
480  throw std::runtime_error("SurfaceMesh is not pseudomanifold. Found an edge connected to more than 2 faces.");
481  }
482  else if (up.size() < 2) // Edge is a boundary
483  {
484  // std::cerr << "This edge participates in fewer than 2 faces. "
485  // << "Returning..." << std::endl;
486  continue;
487  }
488  auto name = mesh.get_name(edgeID);
489  std::pair<Vertex, Vertex> shared;
490  shared.first = *mesh.get_simplex_up({name[0]});
491  shared.second = *mesh.get_simplex_up({name[1]});
492 
493 
494  std::pair<Vertex, Vertex> notShared;
495  notShared.first = *mesh.get_simplex_up({up[0]});
496  notShared.second = *mesh.get_simplex_up({up[1]});
497 
498  // Check if the edge is a part of a tetrahedron.
499  if (mesh.exists<2>({up[0], up[1]}))
500  {
501  // std::cerr << "Found a tetrahedron cannot edge flip."
502  // << std::endl;
503  continue;
504  }
505 
506  // Check if we're on a ridge. This prevents folding also.
507  if (preserveRidges)
508  {
509  auto a = getNormal(mesh, mesh.get_simplex_up(edgeID, up[0]));
510  auto b = getNormal(mesh, mesh.get_simplex_up(edgeID, up[1]));
511  auto val = angle(a, b);
512  if (val > 60)
513  {
514  continue;
515  }
516  }
517 
518  // TODO: (5) Change to check link condition
519  // Check if the triangles make a wedge shape. Flipping this can
520  // cause knife faces.
521  auto f1 = (shared.first - notShared.first)^(shared.second - notShared.first);
522  auto f2 = (shared.first - notShared.second)^(shared.second - notShared.second);
523  auto f3 = (notShared.first - shared.first)^(notShared.second - shared.first);
524  auto f4 = (notShared.first - shared.second)^(notShared.second - shared.second);
525  auto area = std::pow(std::sqrt(f1|f1) + std::sqrt(f2|f2), 2);
526  auto areaFlip = std::pow(std::sqrt(f3|f3) + std::sqrt(f4|f4), 2);
527  if (areaFlip/area > 1.01) continue; // If area changes by a lot then continue
528 
529  // Check the flip using user function
530  if (checkFlip(mesh, edgeID))
531  {
532  *iter++ = edgeID; // Insert into edges to flip
533 
534  // The local topology will be changed by edge flip.
535  // Don't flip edges which share a common face.
536  std::set<SurfaceMesh::SimplexID<2> > tmpIgnored;
537  kneighbors(mesh, edgeID, 3, tmpIgnored);
538  ignoredEdges.insert(tmpIgnored.begin(), tmpIgnored.end());
539  // neighbors(mesh, edgeID, std::inserter(ignoredEdges, ignoredEdges.end()));
540  }
541  }
542  }
543  }
