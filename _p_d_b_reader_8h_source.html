244 static void initElementMap(){
245  // If the element map is empty build it...
246  if(detail::PDBelementMap.size() == 0)
247  {
248  for(int i = 0; i < detail::MAX_BIOCHEM_ELEMENTS; ++i)
249  {
250  const std::string atomName = detail::PDBelementTable[i].atomName;
251  const std::string residueName = detail::PDBelementTable[i].residueName;
252  detail::PDBelementMap[residueName][atomName] = detail::PDBelementTable[i];
253  }
254  }

257 template <typename Inserter>
258 bool readPDB(const std::string& filename, Inserter inserter)
259 {
260  std::ifstream infile(filename);
261  std::string line;
262 
263  initElementMap(); // Init the element map if it
264 
265  if(infile.is_open())
266  {
267  while (std::getline(infile, line))
268  {
269  std::smatch match;
270  if (std::regex_match(line, match, detail::atom))
271  {
272  Atom atom;
273  // See PDB file formatting guidelines
274  float x = std::atof(line.substr(30,8).c_str());
275  float y = std::atof(line.substr(38,8).c_str());
276  float z = std::atof(line.substr(46,8).c_str());
277  atom.pos = Vector({x, y, z});
278 
279  atom.radius = 1.0f; // default radius
280  std::string atomName = line.substr(12,4);
281  std::string residueName = line.substr(17,3);
282 
283  auto innerMapIT = detail::PDBelementMap.find(residueName);
284 
285  if(innerMapIT != detail::PDBelementMap.end()){
286  auto innerMap = innerMapIT->second;
287  auto typeIT = innerMap.find(atomName);
288  if(typeIT != innerMap.end()){
289  atom.radius = typeIT->second.radius;
290  }
291  else{
292  std::cout << "Could not find atomtype of '"
293  << atomName << "' in residue '"
294  << residueName << "'. "
295  << "Using default radius." << std::endl;
296  }
297  }
298  else{
299  std::cout << "Could not find ResidueName '"
300  << residueName << "' in table. "
301  << "Using default radius." << std::endl;
302  }
303  *inserter++ = atom;
304  }
305  }
306  return true;
307  }
308  else
309  {
310  std::cerr << "Unable to open \"" << filename << "\"" << std::endl;
311  return false;
312  }
313 }

316 template <typename Iterator, typename BlurFunc>
317 void getMinMax(Iterator begin, Iterator end, f3Vector& min, f3Vector& max, BlurFunc &&f)
318 {
319  float maxRad = 0.0;
320  float tmpRad;
321 
322  min[0] = min[1] = min[2] = std::numeric_limits<float>::infinity();
323  max[0] = max[1] = max[2] = -std::numeric_limits<float>::infinity();
324 
325  for (auto curr = begin; curr != end; ++curr)
326  {
327  float x = curr->pos[0];
328  float y = curr->pos[1];
329  float z = curr->pos[2];
330 
331  if (min[0] > x)
332  min[0] = x;
333  if (max[0] < x)
334  max[0] = x;
335 
336  if (min[1] > y)
337  min[1] = y;
338  if (max[1] < y)
339  max[1] = y;
340 
341  if (min[2] > z)
342  min[2] = z;
343  if (max[2] < z)
344  max[2] = z;
345 
346  tmpRad = f(curr->radius); // * sqrt(1.0 + log(detail::EPSILON) / blobbyness);
347  if (maxRad < tmpRad)
348  maxRad = tmpRad;
349  }
350 
351  min -= f3Vector({maxRad, maxRad, maxRad});
352  max += f3Vector({maxRad, maxRad, maxRad});
353 }

359  const f3Vector& maxMin,
360  const i3Vector& dim,
361  float blobbyness)
362 {
363 
364  // Functor to calculate gaussian blur
365  auto evalDensity = [blobbyness](const Atom& atom, f3Vector& pnt, float maxRadius)
366  -> float {
367  double expval;
368 
369  f3Vector tmp = atom.pos - pnt;
370  double r = tmp|tmp;
371  double r0 = atom.radius*atom.radius;
372 
373  // expval = BLOBBYNESS*(r/r0 - 1.0);
374  expval = blobbyness*(r-r0);
375 
376  // Truncate gaussian
377  if (sqrt(r) > maxRadius)
378  {
379  return 0.0;
380  }
381  return (float) exp(expval);
382  };
383 
384  f3Vector span;
385  span = (maxMin).ElementwiseDivision(static_cast<f3Vector>((dim - i3Vector({1,1,1}))));
386 
387  float radFactor = sqrt(1.0 + log(detail::EPSILON)/(2.0 * blobbyness));
388 
389  for (auto curr = begin; curr != end; ++curr){
390  float maxRad = curr->radius * radFactor;
391  // compute the dataset coordinates of the atom's center
392  f3Vector tmpVec = (curr->pos-min).ElementwiseDivision(span);
393  i3Vector c;
394  std::transform(tmpVec.begin(), tmpVec.end(), c.begin(), [](float v)-> int {return round(v);});
395 
396  // std::cout << "Max Radius: " << maxRad << std::endl;
397 
398  // compute the bounding box of the atom (maxRad^3)
399  i3Vector amin;
400  i3Vector amax;
401  for (int j = 0; j < 3; ++j){
402  int tmp;
403  float tmpRad = maxRad/span[j];
404 
405  tmp = (int)(c[j] - tmpRad - 1);
406  amin[j] = (tmp < 0) ? 0 : tmp; // check if tmp is < 0
407  tmp = (int)(c[j] + tmpRad + 1);
408  amax[j] = (tmp > (dim[j] - 1)) ? (dim[j] - 1) : tmp;
409  }
410 
411  // std::cout << amin << " " << amax << std::endl;
412 
413  // Blur kernel in bounding box
414  for (int k = amin[2]; k <= amax[2]; k++)
415  {
416  for (int j = amin[1]; j <= amax[1]; j++)
417  {
418  for (int i = amin[0]; i <= amax[0]; i++)
419  {
420  f3Vector pnt = min + f3Vector({static_cast<float>(i),
421  static_cast<float>(j),
422  static_cast<float>(k)}).ElementwiseProduct(span);
423  dataset[Vect2Index(i, j, k, dim)] += evalDensity(*curr, pnt, maxRad);
424  }
425  }
426  }
